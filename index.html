Jak nastavit Google Tabulky

1. Otevři sheets.google.com a vytvoř novou tabulku.

2. V tabulce klikni na Rozšíření → Apps Script.

3. Celý obsah nahraď tímto kódem:

const SHEET_NAME = "Smeny"; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(["id","typ","datum","start","end","poznamka"]); } const action = (e.parameter && e.parameter.action) || "get"; if (action === "add") { sheet.appendRow([ e.parameter.id, e.parameter.typ, e.parameter.datum, e.parameter.start, e.parameter.end, e.parameter.poznamka || "" ]); return out({ok:true}); } if (action === "delete") { const id = e.parameter.id; const data = sheet.getDataRange().getDisplayValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(id)) { sheet.deleteRow(i+1); break; } } return out({ok:true}); } if (action === "update") { const id = e.parameter.id; const data = sheet.getDataRange().getDisplayValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(id)) { sheet.getRange(i+1,1,1,6).setValues([[ id, e.parameter.typ, e.parameter.datum, e.parameter.start, e.parameter.end, e.parameter.poznamka||"" ]]); break; } } return out({ok:true}); } const data = sheet.getDataRange().getDisplayValues(); if (data.length <= 1) return out([]); const rows = data.slice(1).map(r => ({ id:String(r[0]), typ:String(r[1]), datum:String(r[2]), start:String(r[3]), end:String(r[4]), poznamka:String(r[5]) })); return out(rows); } function out(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }

4. Ulož (Ctrl+S), pak Nasadit → Nové nasazení.

5. Typ: Webová aplikace. Přístup: Kdokoliv. Klikni Nasadit.

6. Zkopíruj URL a vlož ji výše.
